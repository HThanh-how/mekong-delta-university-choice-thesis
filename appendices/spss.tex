\chapter{HỒ SƠ PHÂN TÍCH SPSS}
\section*{Mục đích}
Phụ lục trình bày cú pháp SPSS để nhập tệp \texttt{data/survey\_486.csv}, tính điểm thang đo và chạy lại các phân tích chính. Tên biến và định dạng được khai báo tường minh để cú pháp không phụ thuộc vào trình hướng dẫn nhập dữ liệu.

\section*{Cú pháp SPSS tái lập}
\VerbatimInput[fontsize=\scriptsize,breaklines=true,breakanywhere=true]{analysis/spss_analysis.sps}

\section*{Các bảng đầu ra cần lưu từ SPSS}
\begin{itemize}[leftmargin=1.5cm]
\item Frequencies, Descriptives, Reliability Statistics và Item--Total Statistics.
\item KMO and Bartlett's Test, Total Variance Explained và Pattern Matrix.
\item Correlations, Model Summary, ANOVA và Coefficients kèm Tolerance/VIF.
\item Biểu đồ phần dư, Cook's Distance, Independent Samples Test và ANOVA theo nhóm.
\end{itemize}
